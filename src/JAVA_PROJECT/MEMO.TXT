CREATE TABLE users (
    id VARCHAR2(50) NOT NULL,
    username VARCHAR2(50) NOT NULL UNIQUE, -- 중복되지 않도록 고유 제약 조건 추가
    password VARCHAR2(50) NOT NULL,
    PRIMARY KEY (id)
);
select * from users;

CREATE TABLE posts (
    post_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY,
    title VARCHAR2(100),
    content CLOB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (post_id)
);

select * from posts;
